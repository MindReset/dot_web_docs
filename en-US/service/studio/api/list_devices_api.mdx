    ```go
    package main

    import (
      "fmt"
      "net/http"
      "io/ioutil"
    )

    func main() {
      url := "https://dot.mindreset.tech/api/authV2/open/devices"
      req, _ := http.NewRequest("GET", url, nil)
      req.Header.Set("Authorization", "Bearer {{API_KEY}}")
      client := &http.Client{}
      resp, _ := client.Do(req)
      defer resp.Body.Close()
      respBody, _ := ioutil.ReadAll(resp.Body)
      fmt.Println(string(respBody))
    }
    ```

## Troubleshooting

### Endpoint

`/api/authV2/open/devices`

### Method

`GET`

### Request Schema

This interface does not require a request body, only an API key in the request header.

### Response Schema

<TypeTable
	type={{
		'[]': {
			description: 'Device list array',
			type: 'array',
			required: true,
		},
		id: {
			description: 'Device serial number',

			required: true,
		},
		series: {

### Status Codes

| code  | Meaning      | Description                        |
| :---- | :----------- | :--------------------------------- |
| `200` | Success      | Successfully retrieved device list |
| `401` | Unauthorized | Invalid or expired API key         |
| `500` | Server Error | Internal server error              |
